#Configuracion de git:
git --version : muestra la version de git que tenemos instalada.
git config --global user.name [nombre_de_usuario] : configurar el nombre.
git config --global user.email [correo_electronico] : configurar el correo.
git config --global user.editor "code --wait" : configurar el editor de codigo por defecto a vscode.
git config --global init.defaultBranch [nombre_de_la_rama] : configurar nombre de la rama global, sugerencia "main".
git config --global core.autocrlf input : en caso de tener mac o linux.
git config --global core.autocrlf true : en caso de tener windows.
git config --global -e : muestra la configuracion global.
git config -h : lista de todas las configuraciones de git.

#Comandos basicos:
git init : inicializa un nuevo repositorio en git.
git status : muestra el estado actual del repositorio.
git status -s : muestra un mejor git status.
git add : agrega los cambios realizados a una zona de staged.
git commit -m [nombre_del_commit]: guarda los cambios que estan en la zona de staged en el historial de git. 
git push -u origin main : subir los cambios a github.
git diff : ver los cambios.
git diff --staged : ver los cambios que se encuentran en la zona de staged.
git log --oneline : muestra el historial de nuestros commits. 
git clone [url_del_repositorio] : clona el repositorio de la nube a local.

#Ramas
git branch : muestra la rama actual.
git branch -a : muestra todas las ramas /local y remoto.
git branch -m [nombre_de_la_rama] : cambiar nombre de una rama.
git checkout -b [nombre_de_la_rama] : crea una nueva rama.
git checkout [nombre_de_la_rama] : cambia de rama.
git merge [nombre_de_la_rama] : traer cambios de una rama.

#Subir un nuevo repositorio local a remoto
git remote add origin [url_del_repositorio] 
git push -u origin main

#quitar, mover, eliminar, renombrar archivos.
git rm [nombre_del_archivo] : elimina el archivo y lo agrega a la zona de staged.
git restore --cached [nombre_del_archivo] : quita los archivos de la zona de staged.
git restore [nombre_del_archivo] : descarta los cambios.
git mv [nombre_del_archivo] [nuevo_nombre] : renombrar y agregar a staged.

#comandos de la terminal
mkdir [nombre_del_directorio] : crea una carpeta nueva.
touch [nombre_del_archivo] : crea un nuevo archivo.
cat [nombre_del_archivo] : muestra el contenido del archivo.
